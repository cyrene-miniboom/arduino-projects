核心代码展示


#include 
#define I2C_ADDR 0x27
#define LCD_COLUMNS 20
#define LCD_LINES 4

LiquidCrystal_I2C lcd(I2C_ADDR, LCD_COLUMNS, LCD_LINES);

float cm;
float inches;
const float BETA = 3950;

void setup() {
    // Init
    lcd.init();
    lcd.backlight();
    Serial.begin(9600);

    pinMode(7, OUTPUT);
    pinMode(13, OUTPUT);
    pinMode(3, OUTPUT);
    pinMode(2, INPUT);
}

void loop() {
    float celsius = 25;

    float speed = 0.033145 * sqrt((273.15 + celsius) / 273.15);

    digitalWrite(3, HIGH);
    delayMicroseconds(10);
    digitalWrite(3, LOW);

    int duration = pulseIn(2, HIGH);

    float distance = (duration * speed) / 2;

    Serial.print("Distance: ");
    Serial.print(distance);
    Serial.println(" cm");

    if (distance < 20) {
        digitalWrite(13, HIGH);
        tone(7, 261);
        int times = distance * 20;
        delay(times);
        digitalWrite(13, LOW);
        noTone(7);
    }

    lcd.setCursor(0, 0);
    lcd.print("distance:");
    lcd.setCursor(10, 0);
    lcd.print(distance);

    delay(10);
}
